CI Integration

Example GitHub Action to run Flex Audit and upload SARIF:

name: Flex Audit
on: [push, pull_request]

jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - name: Install deps
        run: pip install -r requirements.txt
      - name: Run Flex Audit
        run: python -m flex_audit.src . --format sarif --output ./audit_output --skip-third-party
      - name: Upload SARIF
        uses: github/codeql-action/upload-sarif@v1
        with:
          sarif_file: audit_output/audit_report.sarif
